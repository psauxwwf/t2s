- [ ] Изучи этот проект https://github.com/samosvalishe/free-turn-proxy
      В нем есть клиентская часть, ее бекенд надо добавить как клиента в текущий проект t2s
      Запускается клиент так (через # приведены параметры которые должны задаваться через конфиг):
      ./client-linux-amd64 \
      -listen 127.0.0.1:3080 \ # генерируется сам по аналогии с остальными бекендами (динамичный порт RelayPort int `yaml:"-"`)
      -peer 203.0.113.10:56002 \ # задается через конфиг одной строкой
      -links "https://vk.ru/call/join/TP8BQGL6pZUKcq_wkOzzwkpDpTPl3FawYAl2hS91i-8,https://vk.ru/call/join/TP8BQGL6pZUKcq_wkOz111111111111111111111i-8" \ # задается через конфиг массивом
      -mode tcp \ # всегда tcp добавлять в конфиг не надо
      -obf-profile rtpopus \ # задается строкой в зависимости от исходного кода будет строкой которая возможно мапится в какой-то тип
      -obf-key 61caefa65c98d428b426d40c75251d77d96a46c2c8962284168704ba37ef6ef5 \ # задается через конфиг строкой
      -manual-captcha # bool переменная - задается строкой (по стандарту всегда false)
